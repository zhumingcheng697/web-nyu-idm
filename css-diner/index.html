class Student: Person, Talkable, Walkable {

     public var name: String

     private var classTime: DateInterval

     private var classLocation: CLLocation

     private var instructor: Person

    

     init(name: String, classTime: DateInterval, classLocation: CLLocation, instructor: Person) {

         self.name = name

         self.classTime = classTime

         self.classLocation = classLocation

         self.instructor = instructor

     }

    

     func greetInstructor() {

         self.talk("Hi, \(self.instructor.name)")

     }

    

     func goToClass() {

         let interval = Date().timeIntervalSince(self.classTime.start)

         if interval > 0 {

             self.walk(Pace.fast, to: self.location)

             print("Run fast! You are late!")

         } else if interval < -1800 {

             print("Do you want to leave this early?")

         } else {

             self.walk(Pace.normal, to: self.location)

         }

     }

}
